  \section{\textbf{Problem}}
 Find the area of the region bounded by the circle $\bf{x}^{T}$ $\bf{x}$= 4 and \norm{\vec{x}-\myvec{2\\0}}=2.
  \section{\textbf{Solution}}
  \norm{\vec{x}}^2 + 2\vec{u}^T\vec{x} + f = 0\\
  {\vec{x}^T\vec{x}} + 2\vec{u}^T\vec{x} + f = 0\\
  $So from above equation we can say that,
  \subsection{Circle 1}
  Taking equation of the first circle to be,
 \begin{align}
 \norm{\vec{x}}^2 + 2\vec{u}_1^T\vec{x} + f_1 = 0\label{2.1.1}\\
 {\vec{x}^{T}\vec{x}}-4=0 \label{2.1.2}\text{(given)}\\
 \vec{u_1}=\myvec{0\\0}\\
 f_1=-4
 \end{align}
 
 \subsection{Circle 2}
Taking equation of the second circle to be,
\begin{align}
  \norm{\vec{x}-\myvec{2\\0}}^2=2^2 \text{(given)}\\
  {\vec{x}^{T}\vec{x}+2\vec{u_2}^T\vec{x}}=0 \label{2.2.2}\\
  \vec{u_2}=\myvec{-2\\0}\\
 f_2=0
  \end{align}
Now,Subtracting equation \eqref{2.2.2} from \eqref{2.1.2} We get,
 \begin{align}
 \vec{x}^T\vec{x}-2\vec{u_2}^T\vec{x}+f_1-\vec{x}^{T}\vec{x}=0\\
 2\vec{u}^{T}\vec{x}=-4\\
 \myvec{-4&0}\vec{x}=-4
 \end{align}
 Which can be written as:-
 \begin{align}
 \myvec{1&0}\vec{x}=1\\
 \vec{x}=\myvec{1 \\ 0} + \lambda \myvec{0\\1}\\
\vec{x}=\vec{q}+\lambda\vec{m}\label{2.2.10}\\
\vec{q}=\myvec{1 \\ 0} \\
\vec{m}=\myvec{0\\1}
  \end{align}
 Substituting \eqref{2.2.10} in \eqref{2.1.1}
  \begin{align}
 \norm{\vec{x}}^2 + 2\vec{u}_1^T\vec{x} + f_1 = 0\\
 \norm{\vec{q}+\lambda\vec{m}}^2 + f_1 = 0\\
 \norm{\myvec{1\\\lambda}}^2-4=0\\
 \lambda^2+1-4=0\\
 \lambda=+\sqrt{3},-\sqrt{3}\\
 \end{align}
Substituting the value of $\lambda$ in\eqref{2.2.10}
\begin{align}
\vec{x}=\vec{q}+\lambda\vec{m}\\
For \lambda=\sqrt{3},\vec{x}=\myvec{1\\\sqrt{3}}\\
For \lambda=-\sqrt{3},\vec{x}=\myvec{1\\-\sqrt{3}}
 \end{align} 
 Points of intersection come out to be(1,\sqrt{3}) and(1,-\sqrt{3}),$Now finding the direction vector between point of intersection and origin of two circle.
 Subtracting Point 1 from point 2 direction vector comes out to be k_1\myvec{0\\ \sqrt{3}}
 $Subtracting both the origin then the vector come out to be k_2\myvec{2\\0}
 $Now using these to find angle between two vectors by finding inner product:-\\
\begin{align}
    \vec{a}\vec{b}=\norm{a}\norm{b}\cos\theta\\
    \frac{0}{\norm{a}\norm{b}}=\cos\theta\\
    \theta=90
\end{align}
$\theta$ gives the angle segment. We have to Double that to find out area.
 \begin{align}
Area=Area of sector-Area of Triangle\\
Area=\frac{\pi\theta}{360}r^2-\frac{1}{2}2\sqrt{3}\\
Total area=2*Area\\
=2\pi-2\sqrt{3}
\end{align}
