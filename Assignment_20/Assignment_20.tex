   \section{\textbf{Problem}}
   Let $\mathbb{F}$ be a field and let $f$ be the linear functional on $\mathbb{F}^2$ defined by,
\begin{align*}
f(x_1,x_2) = ax_1+bx_2
\end{align*}
For the linear operator $T(x_1,x_2) = (x_1-x_2,x_1+x_2)$\\
Let, $g = T^ty$ and find $g(x_1,x_2)$
\section{\textbf{Solution}}
The linear functional $f$ on $\mathbb{F}^2$ is defined by,
\begin{align}
f(\vec{x}) &= \vec{a^T}\vec{x}
\quad{\text{$\forall \vec{x} \in \mathbb{F}^2$}}\label{def}
\end{align}
where,
\begin{align}
\vec{a} &= \myvec{a\\b}
\end{align}
We use the following theorem,\\
Let $\mathbb{V}$ and $\mathbb{W}$ be vector spaces, over the field $F$. For each linear transformation $T: \mathbb{V} \xrightarrow{} \mathbb{W}$, there is a unique linear transformation $T^t: \mathbb{W}^* \xrightarrow{} \mathbb{V}^*$ such that,
\begin{align}
(T^tg)(\alpha) &= g(T\alpha)\label{theorem}
\end{align}
$\forall (\vec{x}) \in \mathbb{F}^2$ the given linear operator $T$ defined as,
\begin{align}
T(\vec{x}) &= \vec{A}\vec{x}
\end{align}
Where,
\begin{align}
\vec{A}=\myvec{1&-1\\1&1}
\end{align}
Hence,
\begin{align}
\vec{A}\vec{x}=\myvec{1&-1\\1&1}\vec{x} \label{qn}
\end{align}
Consider the following mapping,
\begin{align}
g = T^tf\label{map}
\end{align}
Now,
\begin{align}
g(\vec{x}) &= T^tf(\vec{x})&\label{ABC}
\end{align}
Using \eqref{theorem} in \eqref{ABC},
\begin{align}
&= f(T(\vec{x}))\\
&= \vec{a^T}\vec{A}\vec{x}\\
&= \vec{a^T}\myvec{1&-1\\1&1}\vec{x}\\
&= a(x_1-x_2)+b(x_1+x_2) \label{ans}
\end{align}
